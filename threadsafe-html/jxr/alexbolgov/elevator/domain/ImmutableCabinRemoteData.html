1   package alexbolgov.elevator.domain;


2   


3   import java.util.Collections;


4   import java.util.HashSet;


5   import java.util.Objects;


6   import java.util.Set;


7   


8   /**


9    * @author Alexander Bolgov on 26.11.2017.


10   */


11  public class ImmutableCabinRemoteData implements CabinRemote.CabinRemoteData {


12  


13      private final int floor;


14      private final Set<Integer> calledFloors;


15      private final ElevatorDirection direction;


16  


17      public ImmutableCabinRemoteData(int floor, final Set<Integer> calledFloors, final ElevatorDirection direction){


18          Objects.requireNonNull(calledFloors);


19          Objects.requireNonNull(direction);


20          this.floor = floor;


21          this.calledFloors = Collections.unmodifiableSet(new HashSet<>(calledFloors));


22          this.direction = ElevatorDirection.valueOf(direction.toString());


23      }


24  


25      @Override


26      public int getFloor() {


27          return floor;


28      }


29  


30      @Override


31      public Set<Integer> getCalledFloors() {


32          return calledFloors;


33      }


34  


35      @Override


36      public ElevatorDirection getDirection() {


37          return direction;


38      }


39  }
