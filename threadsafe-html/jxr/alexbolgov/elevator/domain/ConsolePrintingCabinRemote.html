1   package alexbolgov.elevator.domain;


2   


3   import java.util.HashSet;


4   import java.util.Set;


5   


6   /**


7    * @author Alexander Bolgov on 28.11.2017.


8    */


9   public class ConsolePrintingCabinRemote implements CabinRemote {


10  


11      private final ControlUnit controlUnit;


12  


13      private ElevatorDirection direction = ElevatorDirection.WAITING;


14      private int floor;


15      private Set<Integer> pressedButtons = new HashSet<>();


16  


17      public ConsolePrintingCabinRemote(ControlUnit controlUnit) {


18          this.controlUnit = controlUnit;


19      }


20  


21      @Override


22      public void goToFloor(int floorNumber) {


23          controlUnit.goTo(0, floorNumber);


24      }


25  


26      @Override


27      public void updateData(final CabinRemoteData data) {


28          updateFloor(data.getFloor());


29          updatePressedButtons(data.getCalledFloors());


30          updateDirection(data.getDirection());


31      }


32  


33      private void updateFloor(int floor) {


34          if (this.floor != floor) {


35              this.floor = floor;


36              System.out.println(
"[Elevator Remote] I am on floor "
 + this.floor);


37          }


38      }


39  


40      private void updatePressedButtons(final Set<Integer> pressedButtons) {


41          if (!this.pressedButtons.equals(pressedButtons)) {


42              this.pressedButtons = pressedButtons;


43              System.out.println(
"[Elevator Remote] Pressed buttons: "
 + this.pressedButtons);


44          }


45      }


46  


47      private void updateDirection(final ElevatorDirection direction) {


48          if (this.direction != direction) {


49              this.direction = direction;


50              System.out.println(
"[Elevator Remote] Direction is "
 + this.direction);


51          }


52      }


53  


54  }
