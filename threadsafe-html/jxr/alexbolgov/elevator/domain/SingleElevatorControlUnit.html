1   package alexbolgov.elevator.domain;


2   


3   import com.google.common.collect.Sets;


4   import sun.reflect.generics.reflectiveObjects.NotImplementedException;


5   


6   import java.util.*;


7   import java.util.concurrent.*;


8   import java.util.concurrent.locks.ReadWriteLock;


9   import java.util.concurrent.locks.ReentrantReadWriteLock;


10  


11  import static alexbolgov.elevator.domain.DoorsState.CLOSED;


12  import static alexbolgov.elevator.domain.DoorsState.OPENED;


13  import static alexbolgov.elevator.domain.ElevatorActivity.MOVING;


14  import static alexbolgov.elevator.domain.ElevatorActivity.STOPPED;


15  import static alexbolgov.elevator.domain.ElevatorDirection.*;


16  


17  /**


18   * @author Alexander Bolgov on 20.11.2017.


19   */


20  public class SingleElevatorControlUnit implements ControlUnit {


21  


22      private final int numOfFloors;


23      private final long timeUnitsPerFloor;


24      private final TimeUnit timeUnit;


25      private final long doorCloseDelay;


26      private final TimeUnit doorCloseTimeUnit;


27  


28      private final NavigableSet<Integer> floorStops = new ConcurrentSkipListSet<>();


29      private final NavigableSet<Integer> elevatorStops = new ConcurrentSkipListSet<>();


30      private volatile int currentFloor = 1;


31      private volatile ElevatorDirection direction = WAITING;


32      private volatile ElevatorActivity activity = STOPPED;


33      private volatile DoorsState doors = CLOSED;


34      private volatile ScheduledFuture<?> doorsClosedFuture = null;


35  


36      private final Set<FloorRemote> remotes = Collections.newSetFromMap(new ConcurrentHashMap<>());


37      private volatile Elevator elevator;


38  


39      private final ReadWriteLock lock = new ReentrantReadWriteLock();


40      private final ScheduledExecutorService executorService = new ScheduledThreadPoolExecutor(1);


41  


42      public SingleElevatorControlUnit(


43              int numOfFloors,


44              long timeUnitsPerFloor, final TimeUnit timeUnit,


45              long doorCloseDelay, final TimeUnit doorCloseTimeUnit) {


46          this.numOfFloors = numOfFloors;


47          this.timeUnitsPerFloor = timeUnitsPerFloor;


48          this.timeUnit = timeUnit;


49          this.doorCloseDelay = doorCloseDelay;


50          this.doorCloseTimeUnit = doorCloseTimeUnit;


51      }


52  


53      @Override


54      public void call(int floorNumber) {


55          if (floorNumber <= 0 || floorNumber > numOfFloors) {


56              System.out.println(
"[Control Unit] Not existing floor. Ignoring call"
);


57          } else {


58              if (!floorStops.contains(floorNumber)) {


59                  lock.writeLock().lock();


60                  try {


61                      if (!floorStops.contains(floorNumber)) {


62                          floorStops.add(floorNumber);


63                          updateFloorRemotesData();


64                          startMovingIfStopped(floorNumber);


65                      }


66                  } finally {


67                      lock.writeLock().unlock();


68                  }


69              }


70          }


71      }


72  


73      @Override


74      public void goTo(int elevatorNumber, int floorNumber) {


75          if (floorNumber <= 0 || floorNumber > numOfFloors) {


76              System.out.println(
"[Control Unit] Not existing floor. Ignoring call"
);


77          } else {


78              lock.writeLock().lock();


79              try {


80                  if (needAddStop(floorNumber)) {


81                      elevatorStops.add(floorNumber);


82                      updateElevatorRemoteData(elevator);


83                      startMovingIfStopped(floorNumber);


84                  }


85              } finally {


86                  lock.writeLock().unlock();


87              }


88          }


89      }


90  


91      private boolean needAddStop(int floorNumber) {


92          return (direction == WAITING ||


93                  direction == UP && currentFloor < floorNumber ||


94                  direction == DOWN && floorNumber < currentFloor ||


95                  (activity == STOPPED && direction != WAITING && floorNumber == currentFloor)) &&


96                  !elevatorStops.contains(floorNumber);


97      }


98  


99      @Override


100     public void updateFloor(boolean up) {


101         lock.writeLock().lock();


102         try {


103             if (up) currentFloor++;


104             else currentFloor--;


105 


106             if (currentFloor <= 0 || currentFloor > numOfFloors) {


107                 throw new RuntimeException(
"Elevator crashed!!! =("
);


108             }


109 


110             updateElevatorRemoteData(elevator);


111             updateFloorRemotesData();


112             makeStopIfNeeded();


113         } finally {


114             lock.writeLock().unlock();


115         }


116     }


117 


118     //always called from write lock


119     private void makeStopIfNeeded() {


120         if (elevatorStops.remove(currentFloor) || floorStops.remove(currentFloor)) {


121             updateElevatorRemoteData(elevator);


122             updateFloorRemotesData();


123             final NavigableSet<Integer> allStops = new TreeSet<>(Sets.union(elevatorStops, floorStops));


124             stop();


125 


126             openDoors();


127             //update direction


128             if (direction == UP) {


129                 if (allStops.higher(currentFloor) != null) {


130                     //continue moving


131                 } else if (allStops.lower(currentFloor) != null) {


132                     direction = DOWN;


133                 } else {


134                     direction = WAITING;


135                 }


136             } else if (direction == DOWN) {


137                 if (allStops.lower(currentFloor) != null) {


138                     //continue moving


139                 } else if (allStops.higher(currentFloor) != null) {


140                     direction = UP;


141                 } else {


142                     direction = WAITING;


143                 }


144             } else if (direction == WAITING) {


145                 if (allStops.lower(currentFloor) != null) {


146                     direction = DOWN;


147                 } else if (allStops.higher(currentFloor) != null) {


148                     direction = UP;


149                 }


150             }


151             updateElevatorRemoteData(elevator);


152             closeDoorsAndContinueMoving();


153         }


154     }


155 


156     private void stop() {


157         if (activity != STOPPED) {


158             elevator.stop();


159             activity = STOPPED;


160         }


161     }


162 


163     private void openDoors() {


164         if (doors != OPENED) {


165             elevator.openDoors();


166             doors = OPENED;


167         }


168     }


169 


170     //always called from write lock


171     private void closeDoorsAndContinueMoving() {


172         if (doorsClosedFuture != null && !doorsClosedFuture.isDone()) {


173             doorsClosedFuture.cancel(false);


174         }


175         doorsClosedFuture = executorService.schedule(() -> {


176             lock.writeLock().lock();


177             try {


178                 elevator.closeDoors();


179                 doors = CLOSED;


180                 startMoving();


181             } finally {


182                 lock.writeLock().unlock();


183             }


184         }, doorCloseDelay, doorCloseTimeUnit);


185     }


186 


187     //always called from write lock


188     private void startMoving() {


189         if (direction == UP) {


190             activity = MOVING;


191             goUp();


192         } else if (direction == DOWN) {


193             activity = MOVING;


194             goDown();


195         }


196     }


197 


198     //always called from write lock


199     private void startMovingIfStopped(int floorNumber) {


200         if (activity == STOPPED) {


201             if (direction == WAITING) {


202                 if (floorNumber > currentFloor) {


203                     direction = UP;


204                 } else if (floorNumber < currentFloor) {


205                     direction = DOWN;


206                 } else {


207                     makeStopIfNeeded();


208                 }


209             } else {


210                 makeStopIfNeeded();


211             }


212             updateElevatorRemoteData(elevator);


213             updateFloorRemotesData();


214             if (doors == CLOSED) {


215                 startMoving();


216             }


217         }


218     }


219 


220     private void goDown() {


221         elevator.move(timeUnitsPerFloor, timeUnit, false);


222     }


223 


224     private void goUp() {


225         elevator.move(timeUnitsPerFloor, timeUnit, true);


226     }


227 


228     @Override


229     public void updateFloorRemotesData() {


230         lock.readLock().lock();


231         try {


232             final Map<Integer, Boolean> states = new HashMap<>(numOfFloors);


233             for (int floor = 1; floor <= numOfFloors; ++floor) {


234                 states.put(floor, floorStops.contains(floor));


235             }


236             for (final FloorRemote remote : remotes) {


237                 remote.updateData(new ImmutableFloorRemoteData(currentFloor, direction, states));


238             }


239         } finally {


240             lock.readLock().unlock();


241         }


242     }


243 


244     @Override


245     public void updateElevatorRemoteData(final Elevator elevator) {


246         lock.readLock().lock();


247         try {


248             elevator.getRemote().updateData(new ImmutableCabinRemoteData(currentFloor, elevatorStops, direction));


249         } finally {


250             lock.readLock().unlock();


251         }


252 


253     }


254 


255     @Override


256     public boolean connectFloorRemote(final FloorRemote remote) {


257         return remotes.add(remote);


258     }


259 


260     @Override


261     public boolean disconnectFloorRemote(final FloorRemote remote) {


262         throw new NotImplementedException();


263     }


264 


265     @Override


266     public boolean connectElevator(final Elevator elevator) {


267         if (this.elevator == null) {


268             this.elevator = elevator;


269             updateElevatorRemoteData(elevator);


270             return true;


271         } else {


272             throw new RuntimeException(
"You can plug in only one elevator into SingleElevatorControlUnit"
);


273         }


274     }


275 


276     @Override


277     public boolean disconnectElevator(final Elevator elevator) {


278         throw new NotImplementedException();


279     }
