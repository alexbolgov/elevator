1   package alexbolgov.elevator.domain;


2   


3   import java.util.Collections;


4   import java.util.HashMap;


5   import java.util.Map;


6   


7   /**


8    * @author Alexander Bolgov on 22.11.2017.


9    */


10  public class ImmutableFloorRemoteData implements FloorRemote.FloorRemoteData {


11      private final int elevatorFloor;


12      private final ElevatorDirection elevatorDirection;


13      private final Map<Integer, Boolean> floorButtonStates;


14  


15      public ImmutableFloorRemoteData(int elevatorFloor, ElevatorDirection direction, Map<Integer, Boolean> floorButtonStates) {


16          this.elevatorFloor = elevatorFloor;


17          this.elevatorDirection = ElevatorDirection.valueOf(direction.toString());


18          this.floorButtonStates = Collections.unmodifiableMap(new HashMap<>(floorButtonStates));


19      }


20  


21      @Override


22      public int getElevatorFloor() {


23          return elevatorFloor;


24      }


25  


26      @Override


27      public ElevatorDirection getElevatorDirection() {


28          return elevatorDirection;


29      }


30  


31      @Override


32      public Map<Integer, Boolean> getFloorButtonStates() {


33          return floorButtonStates;


34      }


35  }
